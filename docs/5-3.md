# 저장할 데이터에 맞게 매핑 정의하기
## ✅ 저장할 데이터에 맞게 매핑 정의하기
```
{
    "review_id": 923874239847,
    "user_id": 1024,
    "user_name": "김영희",     
    "categories": ["가전", "청소기"],
    "rating":4.5,
    "is_verified_purchase": true,
    "review_date": "2024-09-21T14:32:00Z",
    "review_text": "배송이 빨랐고 제품도 만족스러워요."
}
```
```
PUT /product_reviews
{
    "mappings": {
        "properties": {
            "review_id": {
                "type": "long"
            },
            "user_id": {
                "type": "long"
            },
            "user_name": {
                "type": "text"
            },
            "categories": {
                "type": "keyword"
            },
            "rating": {
                "type": "double"
            },
            "is_verified_purchase": {
                "type": "boolean"
            },
            "review_date": {
                "type": "date"
            },
            "review_text": {
                "type": "text"
            }
        }
    }
}
```
