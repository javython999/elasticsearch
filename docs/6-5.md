# 특정 조건을 만족하지 않는 데이터를 조회하고 싶을 때(bool: must_not)
## ✅ 특정 조건을 만족하지 않는 데이터를 조회하고 싶을 때(bool: must_not)
```
DELETE /boards

PUT /boards
{
    "mappings": {
        "properties": {
            "board_id": {
                "type": "long"
            },
            "title": {
                "type": "text",
                "analyzer": "nori"
            },
            "category": {
                "type": "keyword"
            },
            "is_notice": {
                "type": "boolean"
            },
            "create_at": {
                "type": "date"
            }
        }
    }
}
```

```
// 광고 게시판 글이 아니면서,
// 공지 글이 아니면서,
// '검색엔진'의 키워드와 관련된 게시글을 조회하고 싶다고 가정하자.
GET /boards/_search
{
    "query": {
        "bool": {
            "must_not": [
              {
                "term": {
                  "category": "광고 게시판"
                }
              }
            ],
            "filter": [
              {
                "term": {
                    "is_notice": false
                }
              }
            ],
            "must": [
              {
                "match": {
                  "title": "검색엔진"
                }
              }
            ]
        }
    }
}


GET /boards/_search
{
    "query": {
        "bool": {
            "must_not": [
              {
                "term": {
                  "category": "광고 게시판"
                }
              }, 
              {
                "term": {
                    "is_notice": true
                }
              }
            ],
            "must": [
              {
                "match": {
                  "title": "검색엔진"
                }
              }
            ]
        }
    }
}
```
